f <- function(x, r, K){
    r * x * (1 - x / K) + x
  }


simulation <- function(r, K = 100, T = 100, x1 = 10){
  
  # Initialize
  x <- numeric(T)
  x[1] <- x1

  # Simulate
  for(t in 1:(T-1)){
    x[t+1] <- f(x = x[t], r, K)
  }
  
  data_frame(time = 1:T,  population = x)
}
simulation(r = 2)  %>%
  ggplot(aes(time, population)) + geom_line()

simulation(r = 2.5)  %>%
  ggplot(aes(time, population)) + geom_line()

K <- 100

r_range <- seq(1.9, 3.5, len = 400)

burnin <- 100
T <- 200
x <- numeric(T)
rsweep <- numeric(T)
state <- K * 0.2

df <- 
purrr::map_df(r_range, function(r){
  
  for(t in 1:burnin){
    state <- f(state, r, K)
  }
  for(t in 1:T){
    state <- f(state, r, K)
    rsweep[t] <- r
    x[t] <- state
  }
  data_frame(r = rsweep, x = x)
  
})
df %>%
  ggplot(aes(r, x)) + geom_point(size = 0.05)
